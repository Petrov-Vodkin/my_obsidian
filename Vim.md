```bash
							"КОМАНДЫ"
Esc			# командный режим 
i			# режим редактирования

:!			# выполнение команды в терминале 
:w file_name# сохранит в текущию dir под именем file_name
:w! dir/name# сохранить в dir
:r file_name# вставляет file_name (вниз от курсора)
:r !ls -l 	# вставляет вывод команды ls -l
:wq 		# сохранение+выход (write quit)
:help w		# СПРАВКА по команде w
Ctrl d		# список возможных команд

u			# отмена 1-го действия
U			# отмена всех изменеий в СТРОКЕ
Ctrl R		# отмена отмены ))) (u || U)

	"Текст"
o	# создаёт пустую СТРОКУ под КУРСОРОМ + режим вставки(написания)	
О	# создаёт пустую строку НАД курсором + режим вставки(написания)	
v	# ВЫДЕЛИТЬ (визуальный режим)
y	# КОПИРОВАТЬ (yw - копировать 1-но слово)
p	# ВСТАВИТЬ (из буфера) ниже курсора (буфер отчищается)

	"Изменения"
i	# режим вставки текста ПЕРЕД - Esc(возврат в обычный) insert
a	# режим добавления текста ПОСЛЕ | - Esc(возврат в обычный) add
A	# режим вставки текста в КОНЦЕ СТРОКИ
R	# замена символов ("затирание"-введённый символ удаляет предыдущий)
r_	# замена 1-го символа (справа от курсора) на _
	"Удаление + Режим вставки"
сe || cw	# удаление до конца СЛОВА НЕ ВКЛЮЧАЯ ПРОБЕЛ + режим вставки 
c$			# удаление от курсора до КОНЦА СТРОКИ + режим вставки 
c^			# удаление от курсора до НАЧАЛА СТРОКИ + режим вставки 
cc			# вырезать СТРОКУ + режим вставки 

	"Удаление"
# цифра перед (x,w,e, dd) указывает их количество def == 1(4x, d10w, 2dd)
x	# удаление СИМВОЛА справа от курсора| X слева
dw	# удаление(вырезать) до конца СЛОВА включая пробел 
de	# удаление(вырезать) до конца СЛОВА НЕ ВКЛЮЧАЯ ПРОБЕЛ 
d$	# удаление(вырезать) от курсора до КОНЦА СТРОКИ
d^	# удаление(вырезать) от курсора до НАЧАЛА СТРОКИ
dd	# вырезать СТРОКУ

	"Поиск \ Замена \ Инфо " 
Ctrl g			# номер строки, инф о файле 
# ПОИСК по тексту
Shift #         # поиск слова пред которым нажмёшь(но это не точнно)  
?искомое слово	# Enter - ПОИСК НАЗАД по тексту
/искомое слово	# Enter - поиск ВПЕРЁД по тексту. +PUSH
	n 			# следующее слово
	Shift N		# предыдущее слово(поиск в обратном направлении - к началу док)
	Ctrl o		# возврат к строке начала поиска| Ctrl i - вперёд
	% 	# курсор на "("  найдёт  ")" и наоборот.Это поиск открытия/закрытия скобок
# ЗАМЕНА
:{пределы}s/{что заменяем}/{на что заменяем}/{опции}
:s/было/стало	# ЗАМЕНА первого в строке совпадения(в строке, где курсор)
		 {опции}
			/g	# замена всех совпадений во всей строке (иначе  первое совпадение в каждой из строк, входящих в {пределы})
		    /c	# замена с запросом на выполнение
 {пределы}
:%s/было/стало	# замена ВО ВСЁМ ФАЙЛЕ первого совпадения в строке 
:.(начать поиск со строки с курсором),$(последняя строка)s/было/стало # точка == строка с курсором
:№,№s/было/стало# №,№ - диапазон строк (с какой, по какою)
# параметры поиска(задать после команды поиска)
:set ic			# игнор РЕгИсТрА
	 hls is		# подсветка результатов поиска
	 is			# подсведка частичных совпадений
   no			# для отключения - добавить перед опцией(noic)
		
	"Перемещение" 
# цифра перед нажатием указывает их количество def == 1, (0 в начало строки)
l # вправо 
k # вверх
j # вниз
h # влево

w # преремещение ВПЕРЁД по НАЧАЛУ слов
e # преремещение ВПЕРЁД по КОНЦУ  слов
b # преремещение НАЗАД  по НАЧАЛУ слов

^ 		# клавиша (переход в начало строки)
$ 		# клавиша (переход в конец строки)
ctrl-b 	# перемещение на один экран назад ('b'ackward)
ctrl-f 	# перемещение на один экран вперед ('f'orward)

gg			# первая строка
Esc num G	# переместиться на номер строки(num) без num -в конец 
num Shift g # переместиться на номер строки(num) без num -в конец файла
Shift G 	# конец файла
Ctrl W 		# перемещение между окнами vim (Alt Tab)

	"Выделить"
g Backspace	# выделить ФРАГМЕНТ (прямоугольный) 
v			# ВЫДЕЛИТЬ (визуальный режим)
v выделяем :w name_file # СОХРАНИТ выделенный ФРАГМЕНТ как name_file

	"Настройка(config)"
vim ~/.vimrc					 # создаём конфиг здесь/.как (~/.vimrc)
:r $VIMRUNTIME/vimrc_exemple.vim # вставляем в созданный фаил пример cfg
```
